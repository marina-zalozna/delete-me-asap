git clean - полная очистка индекса
git log -p - просмотр разницы между раб.каталогом и посл. коммитом
git checkout -b - создание новой ветки и одновременного переключения на нее
git diff - сравнение веток между собой
git branch -d - удаление локальной ветки
git merge - слияние веток
git commit --amend - перезапись посл/коммита.
